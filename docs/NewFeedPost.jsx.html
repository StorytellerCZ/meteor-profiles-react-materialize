/**
 * @class component NewFeedPost
 * @classdesc Component to add new posts to the user feed.
 */
NewFeedPost = React.createClass({
  addPost(e){
    e.preventDefault()

    // get the value
    let body = e.target.postText.value

    // sanitize
    body = sanitizeHtml(body)

    // add post
    Meteor.user().feed().addPost(body)

    // reset the form
    e.target.reset()
  },
  render(){
    if(Meteor.userId()){
      return <form id="postForm" method="post" onSubmit={this.addPost}>
        <div className="input-field">
          <input type="text" className="validate" name="postText" />
          <label htmlFor="postText">New post</label>
        </div>
      </form>
    }
  }
})
